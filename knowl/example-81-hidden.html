\begin{equation*} \vec{y}'' = \begin{pmatrix} 1\amp 1\\ 5\amp -1\end{pmatrix} \vec{y}' + \begin{pmatrix} 2 \amp 1 \\ -5 \amp 2 \end{pmatrix} \vec{y} \end{equation*}
